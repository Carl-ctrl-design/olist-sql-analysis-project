-- ============================================
-- BUSINESS QUESTION 9: Month over Month Revenue Growth
-- How did revenue grow or shrink each month?
-- Tables used: orders, order_payments
-- New concepts: LAG() window function, MoM calculation,
--              growth rate percentage
-- Author: Carl Waiti | April 2026
-- ============================================

-- CTE 1: Calculate total revenue per month
WITH monthly_revenue AS (
    SELECT
        EXTRACT(YEAR FROM o.order_purchase_timestamp)   AS year,
        EXTRACT(MONTH FROM o.order_purchase_timestamp)  AS month,
        ROUND(SUM(op.payment_value), 2)                 AS total_revenue,
        COUNT(DISTINCT o.order_id)                      AS total_orders
    FROM orders o
    JOIN order_payments op
        ON o.order_id = op.order_id
    WHERE o.order_status = 'delivered'
    GROUP BY
        EXTRACT(YEAR FROM o.order_purchase_timestamp),
        EXTRACT(MONTH FROM o.order_purchase_timestamp)
),

-- CTE 2: Use LAG() to bring previous month revenue
-- into the current month row
revenue_with_lag AS (
    SELECT
        year,
        month,
        total_revenue,
        total_orders,

        -- LAG() looks back 1 row and returns that revenue value
        -- OVER (ORDER BY year, month) defines the row order
        -- so LAG knows which row is "previous"
        LAG(total_revenue, 1) OVER (
            ORDER BY year, month
        )                                               AS prev_month_revenue

    FROM monthly_revenue
)

-- FINAL: Calculate MoM growth rate
SELECT
    year,
    month,
    total_revenue,
    total_orders,
    prev_month_revenue,

    -- Revenue difference vs previous month
    ROUND(
        total_revenue - prev_month_revenue
    , 2)                                                AS revenue_change,

    -- Percentage growth vs previous month
    ROUND(
        (total_revenue - prev_month_revenue) /
        NULLIF(prev_month_revenue, 0) * 100
    , 1)                                                AS mom_growth_pct,

    -- Classify each month's performance
    CASE
        WHEN total_revenue > prev_month_revenue THEN 'Growth'
        WHEN total_revenue < prev_month_revenue THEN 'Decline'
        WHEN prev_month_revenue IS NULL         THEN 'First Month'
        ELSE                                         'Flat'
    END                                               AS growth_direction

FROM revenue_with_lag
ORDER BY year ASC, month ASC;
