-- ============================================
-- BUSINESS QUESTION 3: Top 20 Sellers by Revenue
-- Who are the highest performing sellers on Olist?
-- Tables used: order_items, order_payments, 
--              orders, sellers
-- New concepts: Window functions, RANK(), CTE (WITH clause)
-- Author: Carl Waiti | April 2026
-- ============================================

-- STEP 1: We use a CTE (Common Table Expression)
-- A CTE is a temporary named result we build first
-- then query from. Think of it as a temporary table
-- that only exists during this query.

WITH seller_revenue AS (
    SELECT
        s.seller_id,
        s.seller_city,
        s.seller_state,
        ROUND(SUM(oi.price), 2)          AS total_revenue,
        COUNT(DISTINCT oi.order_id)       AS total_orders,
        ROUND(AVG(oi.price), 2)           AS avg_item_price
    FROM order_items oi
    JOIN orders o
        ON oi.order_id = o.order_id
    JOIN sellers s
        ON oi.seller_id = s.seller_id
    WHERE o.order_status = 'delivered'
    GROUP BY
        s.seller_id,
        s.seller_city,
        s.seller_state
)

-- STEP 2: Now query from the CTE and add the RANK
SELECT
    RANK() OVER (ORDER BY total_revenue DESC)  AS revenue_rank,
    seller_id,
    seller_city,
    seller_state,
    total_revenue,
    total_orders,
    avg_item_price
FROM seller_revenue
ORDER BY revenue_rank
LIMIT 20;
